```java
@Service
public class RuleService {
    
    @Autowired
    private RuleRepository ruleRepository;
    
    @Autowired
    private RuleCache ruleCache;
    
    /**
     * 创建规则
     */
    public String createRule(FilterRule rule) {
        rule.setRuleId(UUID.randomUUID().toString());
        rule.setCreateTime(LocalDateTime.now());
        
        // 验证规则定义的合法性
        validateRuleDefinition(rule);
        
        // 保存到数据库
        FilterRule saved = ruleRepository.save(rule);
        
        // 清除缓存
        ruleCache.evictAll();
        
        return saved.getRuleId();
    }
    
    /**
     * 更新规则
     */
    public void updateRule(FilterRule rule) {
        rule.setUpdateTime(LocalDateTime.now());
        
        validateRuleDefinition(rule);
        
        ruleRepository.save(rule);
        ruleCache.evictAll();
    }
    
    /**
     * 删除规则
     */
    public void deleteRule(String ruleId) {
        // 检查该规则是否被使用
        Long usageCount = ruleRepository.countByRuleId(ruleId);
        if (usageCount > 0) {
            throw new BusinessException("规则正在被使用，无法删除");
        }
        
        ruleRepository.deleteById(ruleId);
        ruleCache.evictAll();
    }
    
    /**
     * 获取规则（带缓存）
     */
    @Cacheable(value = "rules", key = "#ruleId")
    public FilterRule getRule(String ruleId) {
        return ruleRepository.findById(ruleId).orElse(null);
    }
    
    /**
     * 查询所有规则
     */
    public List<FilterRule> listRules(String ruleDimension, 
                                     String ruleType) {
        return ruleRepository.findByDimensionAndType(ruleDimension, ruleType);
    }
    
    /**
     * 验证规则定义
     */
    private void validateRuleDefinition(FilterRule rule) {
        RuleDefinition def = rule.getRuleDefinition();
        
        if (def == null || def.getConditions().isEmpty()) {
            throw new BusinessException("规则条件不能为空");
        }
        
        for (RuleCondition condition : def.getConditions()) {
            if (StringUtils.isEmpty(condition.getFieldName())) {
                throw new BusinessException("字段名不能为空");
            }
            
            if (StringUtils.isEmpty(condition.getOperator())) {
                throw new BusinessException("操作符不能为空");
            }
            
            // 验证值
            validateConditionValue(condition);
        }
    }
}
```

**规则执行引擎**

```java
@Service
public class RuleEngineService {
    
    @Autowired
    private RuleService ruleService;
    
    @Autowired
    private RuleCache ruleCache;
    
    /**
     * 根据规则列表构建Spark SQL WHERE子句
     */
    public String buildFilterSQL(List<String> ruleIds, 
                                 String filterLogic) throws Exception {
        List<FilterRule> rules = ruleIds.stream()
            .map(ruleService::getRule)
            .collect(Collectors.toList());
        
        List<String> conditionList = new ArrayList<>();
        
        for (FilterRule rule : rules) {
            String ruleSQL = buildRuleSQL(rule);
            conditionList.add("(" + ruleSQL + ")");
        }
        
        // 使用指定的逻辑关系连接所有规则
        String operator = "AND".equals(filterLogic) ? " AND " : " OR ";
        String finalSQL = String.join(operator, conditionList);
        
        return finalSQL;
    }
    
    /**
     * 构建单个规则的SQL条件
     */
    private String buildRuleSQL(FilterRule rule) {
        RuleDefinition def = rule.getRuleDefinition();
        List<String> sqlConditions = new ArrayList<>();
        
        for (RuleCondition condition : def.getConditions()) {
            String sqlCondition = buildConditionSQL(condition);
            sqlConditions.add(sqlCondition);
        }
        
        String operator = "AND".equals(def.getLogicOperator()) ? " AND " : " OR ";
        return String.join(operator, sqlConditions);
    }
    
    /**
     * 构建单个条件的SQL
     */
    private String buildConditionSQL(RuleCondition condition) {
        String fieldName = escapeFieldName(condition.getFieldName());
        String operator = condition.getOperator();
        
        switch (operator) {
            case "=":
                return fieldName + " = '" + condition.getValue() + "'";
            case "!=":
                return fieldName + " != '" + condition.getValue() + "'";
            case "<":
                return fieldName + " < " + condition.getValue();
            case ">":
                return fieldName + " > " + condition.getValue();
            case "<=":
                return fieldName + " <= " + condition.getValue();
            case ">=":
                return fieldName + " >= " + condition.getValue();
            case "IN":
                String inValues = condition.getValues().stream()
                    .map(v -> "'" + v + "'")
                    .collect(Collectors.joining(","));
                return fieldName + " IN (" + inValues + ")";
            case "NOT_IN":
                String notInValues = condition.getValues().stream()
                    .map(v -> "'" + v + "'")
                    .collect(Collectors.joining(","));
                return fieldName + " NOT IN (" + notInValues + ")";
            case "LIKE":
                return fieldName + " LIKE '%" + condition.getValue() + "%'";
            case "IS_NULL":
                return fieldName + " IS NULL";
            case "IS_NOT_NULL":
                return fieldName + " IS NOT NULL";
            default:
                throw new IllegalArgumentException("不支持的操作符: " + operator);
        }
    }
    
    /**
     * 在Spark中执行规则过滤
     */
    public Dataset<Row> applyFilterRules(Dataset<Row> rpDataFrame,
                                        List<String> ruleIds,
                                        String filterLogic) throws Exception {
        if (ruleIds == null || ruleIds.isEmpty()) {
            return rpDataFrame;
        }
        
        String filterSQL = buildFilterSQL(ruleIds, filterLogic);
        Dataset<Row> filtered = rpDataFrame.filter(filterSQL);
        
        return filtered;
    }
}
```

```java
@Service
public class ReconciliationService {
    
    @Autowired
    private ReconciliationTaskRepository taskRepository;
    
    @Autowired
    private ReconciliationConfigRepository configRepository;
    
    @Autowired
    private LivyService livyService;
    
    @Autowired
    private SparkJobBuilder sparkJobBuilder;
    
    @Autowired
    private RuleEngineService ruleEngineService;
    
    @Autowired
    private ResultService resultService;
    
    @Autowired
    private ReconciliationResultRepository resultRepository;
    
    /**
     * 启动对账任务
     */
    public String startReconciliation(String configId) throws Exception {
        // 1. 验证配置
        ReconciliationConfig config = configRepository.findById(configId);
        validateConfig(config);
        
        // 2. 创建任务记录
        ReconciliationTask task = new ReconciliationTask();
        task.setTaskId(UUID.randomUUID().toString());
        task.setTaskName(config.getTaskName() + "_" + 
                        LocalDateTime.now().format(DateTimeFormatter.ofPattern("yyyyMMddHHmmss")));
        task.setConfigId(configId);
        task.setStatus("PENDING");
        task.setStartTime(LocalDateTime.now());
        task.setCreateTime(LocalDateTime.now());
        
        ReconciliationTask savedTask = taskRepository.save(task);
        
        // 3. 构建Spark作业
        String sparkJobCode = sparkJobBuilder.buildReconciliationJob(config, savedTask);
        
        // 4. 通过Livy提交到Spark
        LivyBatchResponse livyResponse = livyService.submitBatch(sparkJobCode);
        
        savedTask.setLivyBatchId(livyResponse.getBatchId());
        savedTask.setLivySessionUrl(livyResponse.getSessionUrl());
        savedTask.setStatus("RUNNING");
        taskRepository.save(savedTask);
        
        // 5. 异步监控任务状态
        asyncMonitorTask(savedTask.getTaskId());
        
        return savedTask.getTaskId();
    }
    
    /**
     * 异步监控任务状态
     */
    @Async
    public void asyncMonitorTask(String taskId) {
        ReconciliationTask task = taskRepository.findById(taskId);
        
        while (true) {
            try {
                Thread.sleep(5000);  // 每5秒检查一次
                
                LivyBatchStatus status = livyService.getBatchStatus(
                    task.getLivyBatchId());
                
                if ("success".equals(status.getState())) {
                    handleTaskSuccess(taskId, status);
                    break;
                } else if ("error".equals(status.getState()) || 
                          "dead".equals(status.getState())) {
                    handleTaskFailure(taskId, status);
                    break;
                }
            } catch (Exception e) {
                log.error("监控任务失败: " + taskId, e);
            }
        }
    }
    
    /**
     * 处理任务成功
     */
    private void handleTaskSuccess(String taskId, LivyBatchStatus status) 
            throws Exception {
        ReconciliationTask task = taskRepository.findById(taskId);
        
        // 获取Spark作业结果
        String resultPath = status.getLogLines().stream()
            .filter(line -> line.contains("RESULT_PATH:"))
            .findFirst()
            .map(line -> line.substring(line.indexOf("RESULT_PATH:") + 12).trim())
            .orElseThrow(() -> new Exception("无法获取结果路径"));
        
        // 解析结果文件
        ReconciliationSummary summary = resultService.parseReconciliationResult(
            resultPath);
        
        // 更新任务信息
        task.setStatus("SUCCESS");
        task.setEndTime(LocalDateTime.now());
        task.setDuration(Duration.between(task.getStartTime(), 
                                         LocalDateTime.now()).toMillis());
        task.setResultPath(resultPath);
        task.setSourceRecordCount(summary.getSourceTotal());
        task.setRpRecordCountBefore(summary.getRpBeforeFilterTotal());
        task.setRpRecordCountAfter(summary.getRpAfterFilterTotal());
        task.setDiff1Count(summary.getDiff1Count());
        task.setDiff2Count(summary.getDiff2Count());
        task.setDiff3Count(summary.getDiff3Count());
        task.setDiff4Count(summary.getDiff4Count());
        
        taskRepository.save(task);
        
        // 导入结果到数据库
        resultService.importReconciliationResults(taskId, resultPath);
    }
    
    /**
     * 处理任务失败
     */
    private void handleTaskFailure(String taskId, LivyBatchStatus status) {
        ReconciliationTask task = taskRepository.findById(taskId);
        
        task.setStatus("FAILED");
        task.setEndTime(LocalDateTime.now());
        task.setDuration(Duration.between(task.getStartTime(), 
                                         LocalDateTime.now()).toMillis());
        
        String errorMsg = status.getLogLines().stream()
            .filter(line -> line.contains("ERROR") || line.contains("Exception"))
            .collect(Collectors.joining("\n"));
        task.setErrorMsg(errorMsg);
        
        taskRepository.save(task);
    }
}
```

```java
@Service
public class SparkJobBuilder {
    
    @Autowired
    private FileService fileService;
    
    @Autowired
    private RuleEngineService ruleEngineService;
    
    @Autowired
    private FieldMappingUtil fieldMappingUtil;
    
    /**
     * 构建对账Spark作业代码
     */
    public String buildReconciliationJob(ReconciliationConfig config,
                                        ReconciliationTask task) throws Exception {
        StringBuilder jobCode = new StringBuilder();
        
        // 1. Spark初始化
        jobCode.append(buildSparkInit());
        
        // 2. 加载文件
        jobCode.append(buildFileLoading(config, task));
        
        // 3. 应用Filter规则
        jobCode.append(buildFilterRules(config));
        
        // 4. 对账逻辑
        jobCode.append(buildReconciliationLogic(config, task));
        
        // 5. 结果输出
        jobCode.append(buildResultOutput(config, task));
        
        return jobCode.toString();
    }
    
    /**
     * Spark初始化代码
     */
    private String buildSparkInit() {
        return """
            from pyspark.sql import SparkSession
            from pyspark.sql import functions as F
            from pyspark.sql.window import Window
            import json
            from datetime import datetime
            
            spark = SparkSession.builder \\
                .appName("FileReconciliation") \\
                .config("spark.sql.adaptive.enabled", "true") \\
                .config("spark.sql.adaptive.skewJoin.enabled", "true") \\
                .config("spark.sql.shuffle.partitions", "200") \\
                .getOrCreate()
            
            spark.sparkContext.setLogLevel("WARN")
            """;
    }
    
    /**
     * 文件加载代码
     */
    private String buildFileLoading(ReconciliationConfig config,
                                   ReconciliationTask task) throws Exception {
        UploadFile sourceFile = fileService.getFile(config.getSourceFileId());
        UploadFile rpFile = fileService.getFile(config.getRpFileId());
        
        return String.format("""
            # 加载源文件
            source_df = spark.read.parquet("%s")
            source_count = source_df.count()
            print(f"Source file loaded. Records: {source_count}")
            
            # 加载RP文件（保留未过滤版本）
            rp_df_unfiltered = spark.read.parquet("%s")
            rp_count_before = rp_df_unfiltered.count()
            print(f"RP file loaded. Records before filter: {rp_count_before}")
            
            # 缓存数据框以提高性能
            source_df = source_df.cache()
            rp_df_unfiltered = rp_df_unfiltered.cache()
            """, sourceFile.getStorePath(), rpFile.getStorePath());
    }
    
    /**
     * Filter规则应用代码
     */
    private String buildFilterRules(ReconciliationConfig config) throws Exception {
        if (!config.getEnableFilter() || config.getFilterRuleIds().isEmpty()) {
            return """
                # 未启用Filter规则
                rp_df = rp_df_unfiltered
                """;
        }
        
        String filterSQL = ruleEngineService.buildFilterSQL(
            config.getFilterRuleIds(),
            config.getFilterLogic());
        
        return String.format("""
            # 应用Filter规则
            rp_df = rp_df_unfiltered.filter(\"\"\" %s \"\"\")
            rp_count_after = rp_df.count()
            filtered_count = rp_count_before - rp_count_after
            print(f"RP file after filter. Records: {rp_count_after}, Filtered: {filtered_count}")
            
            # 缓存过滤后的数据框
            rp_df = rp_df.cache()
            """, filterSQL);
    }
    
    /**
     * 对账逻辑代码
     */
    private String buildReconciliationLogic(ReconciliationConfig config,
                                           ReconciliationTask task) throws Exception {
        String primaryKey = config.getPrimaryKey();
        Map<String, String> fieldMapping = config.getFieldMapping();
        List<String> compareFields = config.getCompareFields();
        
        // 生成字段映射的join条件
        String joinCondition = buildJoinCondition(primaryKey, fieldMapping);
        
        // 生成字段对比条件
        String mismatchCondition = buildMismatchCondition(
            compareFields, fieldMapping);
        
        return String.format("""
            # ========== 差异识别 ==========
            
            # 差异1: 底层有，RP无
            diff1_df = source_df.join(
                rp_df.select("%s"),
                on="%s",
                how="leftanti"
            )
            diff1_count = diff1_df.count()
            print(f"Diff1 (Source has, RP missing): {diff1_count}")
            
            # 细分原因：判断RP未过滤版本中是否存在
            diff1_with_reason = diff1_df.join(
                rp_df_unfiltered.select("%s").withColumnRenamed("%s", "{pk}_rp"),
                on=F.col("%s") == F.col("{pk}_rp"),
                how="left"
            ).withColumn(
                "reason",
                F.when(F.col("{pk}_rp").isNull(), "MISSING_IN_RP")
                 .otherwise("FILTERED_BY_RULE")
            ).select(
                "%s",
                "reason"
            )
            
            # 差异2: 底层无，RP有
            diff2_df = rp_df.join(
                source_df.select("%s"),
                on="%s",
                how="leftanti"
            )
            diff2_count = diff2_df.count()
            print(f"Diff2 (RP has, Source missing): {diff2_count}")
            
            # 差异3和4: 都有的情况
            joined_df = source_df.join(
                rp_df,
                on=%s,
                how="inner"
            )
            
            # 差异3: 字段不一致
            diff3_df = joined_df.filter(\"\"\" %s \"\"\")
            diff3_count = diff3_df.count()
            print(f"Diff3 (Mismatch fields): {diff3_count}")
            
            # 差异4: 完全一致
            diff4_df = joined_df.join(
                diff3_df.select("%s"),
                on="%s",
                how="leftanti"
            )
            diff4_count = diff4_df.count()
            print(f"Diff4 (Fully match): {diff4_count}")
            """,
            primaryKey, primaryKey,
            primaryKey, primaryKey,
            primaryKey,
            primaryKey,
            primaryKey, primaryKey,
            joinCondition,
            mismatchCondition,
            primaryKey, primaryKey);
    }
    
    /**
     * 结果输出代码
     */
    private String buildResultOutput(ReconciliationConfig config,
                                    ReconciliationTask task) {
        return String.format("""
            # ========== 结果输出 ==========
            
            # 创建结果目录
            result_dir = "/reconciliation/results/%s"
            
            # 保存各类差异
            diff1_with_reason.write.mode("overwrite").parquet(result_dir + "/diff1")
            diff2_df.write.mode("overwrite").parquet(result_dir + "/diff2")
            diff3_df.write.mode("overwrite").parquet(result_dir + "/diff3")
            diff4_df.write.mode("overwrite").parquet(result_dir + "/diff4")
            
            # 生成汇总统计
            summary = {
                "taskId": "%s",
                "source_total": source_count,
                "rp_before_filter": rp_count_before,
                "rp_after_filter": rp_count_after,
                "diff1": diff1_count,
                "diff1_reason_a": diff1_with_reason.filter("reason='MISSING_IN_RP'").count(),
                "diff1_reason_b": diff1_with_reason.filter("reason='FILTERED_BY_RULE'").count(),
                "diff2": diff2_count,
                "diff3": diff3_count,
                "diff4": diff4_count,
                "match_rate": round((diff4_count / source_count * 100), 2) if source_count > 0 else 0
            }
            
            with open("/tmp/reconciliation_summary_%s.json", "w") as f:
                json.dump(summary, f)
            
            print(f"RESULT_PATH:{result_dir}")
            """, task.getTaskId(), task.getTaskId(), task.getTaskId());
    }
    
    /**
     * 生成Join条件
     */
    private String buildJoinCondition(String primaryKey, 
                                     Map<String, String> fieldMapping) {
        String[] keys = primaryKey.split(",");
        return Arrays.stream(keys)
            .map(key -> {
                String mappedKey = fieldMapping.getOrDefault(key.trim(), key.trim());
                return String.format("(F.col('source_df.%s') == F.col('rp_df.%s'))",
                    key.trim(), mappedKey);
            })
            .collect(Collectors.joining(" & "));
    }
    
    /**
     * 生成不一致判断条件
     */
    private String buildMismatchCondition(List<String> compareFields,
                                         Map<String, String> fieldMapping) {
        return compareFields.stream()
            .map(field -> {
                String mappedField = fieldMapping.getOrDefault(field, field);
                return String.format(
                    "CAST(source_df.%s AS STRING) != CAST(rp_df.%s AS STRING)",
                    field, mappedField);
            })
            .collect(Collectors.joining(" OR "));
    }
}
```

```java
        List<ReconciliationResult> results = new ArrayList<>();
        for (Map<String, Object> record : records) {

            result.setTaskId(taskId);
            result.setDiffType("DIFF_1");
            result.setPrimaryKey(record.get("pk_key").toString());
            result.setReason(record.get("reason").toString());
            result.setDetail(record);
            result.setCreateTime(LocalDateTime.now());
            
            results.add(result);

        resultRepository.saveAll(results);
    }

            headerRow.createCell(2).setCellValue("源值");

        for (ReconciliationResult result : results.getContent()) {
            Row row = sheet.createRow(rowIndex++);
            row.createCell(0).setCellValue(result.getPrimaryKey());
            
            if ("DIFF_1".equals(diffType)) {
                row.createCell(1).setCellValue(result.getReason());
                row.createCell(2).setCellValue(result.getDetail().toString());
            }
        }
    }

        Row row = sheet.createRow(rowIndex);
        row.createCell(0).setCellValue(col1);
        row.createCell(1).setCellValue(col2);
    }
```

```html
        <el-button 
          type="success" 
          @click="startUpload"
          :loading="uploading"
        >
          开始上传

<script>
import { uploadFile, uploadChunk } from '@/api/file'

export default {
  data() {
    return {
      uploadForm: {
        fileType: 'SOURCE',
        format: 'CSV',
        selectedFile: null
      },
      uploadProgress: {
        percentage: 0,
        message: ''
      },
      uploading: false,
      chunkSize: 5 * 1024 * 1024  // 5MB
    }
  },
  methods: {
    handleFileSelect(file) {
      this.uploadForm.selectedFile = file
    },
    async startUpload() {
      const file = this.uploadForm.selectedFile
      if (!file) {
        this.$message.error('请先选择文件')
        return
      }

      this.uploading = true
      
      try {
        // 初始化上传
        const fileId = await uploadFile({
          fileName: file.name,
          fileSize: file.size,
          fileType: this.uploadForm.fileType,
          format: this.uploadForm.format
        })

        // 分片上传
        const totalChunks = Math.ceil(file.size / this.chunkSize)
        for (let i = 0; i < totalChunks; i++) {
          const start = i * this.chunkSize
          const end = Math.min(start + this.chunkSize, file.size)
          const chunk = file.slice(start, end)

          await uploadChunk(fileId, i + 1, chunk)

          this.uploadProgress.percentage = Math.round(((i + 1) / totalChunks) * 100)
          this.uploadProgress.message = `上传中... ${this.uploadProgress.percentage}%`
        }

        this.$message.success('文件上传成功')
        this.resetForm()
      } catch (error) {
        this.$message.error('文件上传失败: ' + error.message)
      } finally {
        this.uploading = false
      }
    },
    resetForm() {
      this.uploadForm = {
        fileType: 'SOURCE',
        format: 'CSV',
        selectedFile: null
      }
      this.uploadProgress = {
        percentage: 0,
        message: ''
      }
    }
  }
}
```
